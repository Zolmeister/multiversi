function deepCopy(grid) {
  var s = [];
  if (!grid[0] || !grid[0][0])
    return [];
  for (var i = 0; i < grid.length; i++) {
    var t = []
    for (var j = 0; j < grid[0].length; j++) {
      t.push(grid[i][j]);
    }
    s.push(t);
  }
  return s;
}
