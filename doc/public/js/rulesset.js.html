RulesSet.prototype.newBoard = function() {
  var grid = new Array(this.width);
  for (var i = 0; i < this.width; i++) {
    grid[i] = new Array(this.height);
    for (var j = 0; j < this.height; j++) {
      if (j == 0 && (i < 2 || i > 6)) {
        grid[i][j] = -2;

      } else if ((j == 1 || j == 5 || j == 6) && (i == 0 || i == 8)) {
        grid[i][j] = -2;

      } else if (j == 7 && !(i == 3 || i == 5)) {
        grid[i][j] = -2;

      } else {
        grid[i][j] = -1;
      }
    }
  }
  return grid;
}
